# CDN Links

Production:

```html
<script crossorigin src="https://unpkg.com/@inlet/react-pixi/dist/react-pixi.umd.js"></script>
```

Development:

```html
<script crossorigin src="https://unpkg.com/@inlet/react-pixi/dist/react-pixi.cjs-dev.js"></script>
```

Example with dependencies:

```html
<html>
  <head>
    <script crossorigin src="https://unpkg.com/react@18/umd/react.production.min.js"></script>
    <script crossorigin src="https://unpkg.com/react-dom@18/umd/react-dom.production.min.js"></script>
    <script crossorigin src="https://unpkg.com/pixi.js/dist/pixi.min.js"></script>
    <script crossorigin src="https://unpkg.com/@inlet/react-pixi/dist/react-pixi.umd.js"></script>
    <script crossorigin src="https://unpkg.com/babel-standalone@6/babel.min.js"></script>
  </head>
  <body>
    <div id="root"></div>

    <script type="text/babel">
      const { Stage, Sprite } = ReactPixi

      const App = () => (
        <Stage>
          <Sprite texture={PIXI.Texture.WHITE} tint={0x00ff00} />
        </Stage>
      )
      const container = document.getElementById('root');
      const root = ReactDOM.createRoot(container);
      root.render(<App />);
    </script>
  </body>
</html>
```
